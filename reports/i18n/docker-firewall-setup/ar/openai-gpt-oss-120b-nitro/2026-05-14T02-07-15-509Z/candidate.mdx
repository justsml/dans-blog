## إعداد جدار الحماية لمضيف Docker

1. يُفترض أن النظام هو خادم Debian/Ubuntu  
1. صُمم للعمل على خادم مضيف Docker  

### تثبيت المتطلبات

~~~sh
# جدار حماية شامل مطلوب
apt-get update && apt-get install -y ufw nmap curl
~~~

### الحصول على عناوين IP الداخلية والخارجية

~~~sh
# احصل على عناوين IP الخاصة بك، إخراج بسيط:
hostname --all-ip-addresses

# أو استخدم أداة ip، مثال:
ip addr
~~~

### إعداد جدار الحماية (UFW) - أوامر مثال

~~~sh
ufw logging on # on=low - قد يكون medium أفضل للتشخيص
ufw logging medium
# أولاً، احظر كل شيء
ufw default deny incoming

# مطلوب: اختر *واحدًا* من قواعد الخروج الافتراضية التالية:
ufw default deny outgoing
ufw default allow outgoing
~~~

# السماح وتسجيل جميع اتصالات SSH الجديدة،  
ufw allow log proto tcp from any to any port 22
## السماح بحركة مرور HTTP (بدون تسجيل صريح)
ufw allow out on docker0 53/udp to 172.17.0.1/16
ufw allow out on eth0 to any port 53
ufw allow out on eth0 from 0.0.0.0/0 to any port 80 proto tcp
ufw allow out on eth0 from 0.0.0.0/0 to any port 443 proto tcp

# تفصيلي: ufw allow proto tcp from any to any port 80
ufw allow 80/tcp
ufw allow 443/tcp
ufw allow log 22/tcp
ufw limit ssh # حد أساسي لتقليل هجمات القوة الغاشمة على SSH (4 محاولات)

# ضبط عنوان IP الخارجي الخاص بك
export EXTERNAL_IP=123.123.123.123
# تحديث عنوان IP الخاص بـ Docker إذا لزم الأمر
export DOCKER_IP=172.17.42.1
# تحويل حركة مرور TCP على المنفذ 8080 إلى التطبيق داخل Docker
ufw allow proto tcp from $EXTERNAL_IP port 8080 to $DOCKER_IP port 3000
~~~

## تمكين / بدء جدار الحماية

> كن حذرًا، لا تقفل منفذ SSH الخاص بك (sshd يستخدم المنفذ 22 افتراضيًا)

~~~sh
ufw --force enable

ufw reset
~~~

### اختبار جدار الحماية الخاص بك

> مهم: استخدم عنوان IP/موقع بعيد  

~~~sh
# Verify dependency
apt-get update && apt-get install -y nmap

# Set scan target
export TARGET_HOST=123.123.123.123

# Example Scan Commands:
# Fast open port check
nmap -p 1-10240,27017 -T5 $TARGET_HOST
# Thorough scan
nmap -p 1-10240,27017 --open -v -APN $TARGET_HOST
# Svc Inspection
nmap -p 1-10240,27017 -O --osscan-guess $TARGET_HOST
~~~

> تم! الآن يجب أن ترى فقط المنافذ التي قمت بتكوينها!
