    ```js
      console.log('Done');
    }
    ```

    ```js
    import { Transform } from 'stream';
    const upperCase = new Transform({
      transform(chunk, encoding, callback) {
        callback(null, chunk.toString().toUpperCase());
      }
    });
    process.stdin
      .pipe(upperCase)

    // Input: "hello world"
    ```

  <div className="explanation">
    Les flux de transformation modifient les données lorsqu'elles les traversent. Ici, chaque morceau est :
    1. Converti en chaîne
    2. Transformé en majuscules
    3. Transmis à stdout

    Cela crée un pipeline qui convertit toute l'entrée en majuscules.

    [En savoir plus sur les flux de transformation](https://nodejs.org/api/stream.html#class-streamtransform)

    - Le package `chokidar`
    - Le debouncing du callback
    - `fs.watchFile()` (bien que moins efficace)

    ```js
    buf1.equals(buf2)  // true
    ```

  <div className="question">
    Quel est le principal objectif de la contre-pression des flux ?
  </div>
  </slot>
  <slot name='explanation'>
  <div className="explanation">
    La contre-pression est un mécanisme qui empêche le débordement mémoire en mettant en pause la lecture lorsque la sortie d'écriture ne peut pas suivre.

    Exemple de contre-pression manuelle :
    ```js
    readable.on('data', (chunk) => {
      const canContinue = writable.write(chunk);
      if (!canContinue) {
        readable.pause();
        writable.once('drain', () => readable.resume());
      }
    });
    ```
    `pipe()` gère cela automatiquement !

    [En savoir plus sur la contre-pression](https://nodejs.org/en/learn/modules/backpressuring-in-streams#backpressuring-in-streams)
  </div>
  </slot>
