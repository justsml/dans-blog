    {text: 'null'},
  ]}
>
  <slot name="question">
  <div className="question">
    ¿Qué imprime el siguiente código?
    ```js
    const promises = [
      Promise.resolve('success'),
      Promise.reject('error')
    ];
    Promise.allSettled(promises).then(results => {
      console.log(results[0].status + ': ' + results[0].value);
    });
    ```
  </div>
  </slot>
  <slot name='explanation'>
  <div className="explanation">
    `Promise.allSettled` no se detiene ante el primer error; devuelve un arreglo de objetos detallando el desenlace de cada promesa. Dado que la primera se cumple (`fulfilled`) con el valor `'success'`, la salida en consola será `fulfilled: success`.
  </div>
  </slot>
</Challenge>
