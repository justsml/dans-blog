    ```jsx
    function Counter() {
      const [count, setCount] = useState(0);

      useEffect(() => {
        const id = setInterval(() => {
          console.log('count is', count);
        }, 1000);
        return () => clearInterval(id);
      }, []); // खाली डिप्स

      </button>;
    }
    ```

    क्लासिक स्टेल क्लोज़र। `useEffect` एक बार चलता है (खाली डिप एरे) और एक इंटरवल बनाता है जो माउंट समय पर `count` पर क्लोज़ करता है: `0`। आप चाहे कितनी भी बार क्लिक करें, इंटरवल का क्लोज़र अपडेटेड वैल्यू कभी नहीं देखता।

    बटन लेबल सही से अपडेट होता है — वह फ्रेश रेंडर है। लेकिन `console.log` हमेशा के लिए `0` पर फंसा रहता है।

    इसे ठीक करने के लिए `count` को डिप एरे में जोड़ें (इंटरवल को दोबारा बनाता है) या ref का उपयोग करें:
    ```jsx
    const countRef = useRef(count);
    useEffect(() => { countRef.current = count; }, [count]);
    // फिर इंटरवल के अंदर countRef.current का उपयोग करें
    ```

    ```javascript
      const onClick = () => {
        console.log('clicked!');
      };

      return () => element.removeEventListener('click', onClick);
    }
    ```

  <slot name="question">
  <div className="question">
    `expensive()` वास्तव में कितनी बार कॉल होता है?
    ```javascript
    function memoize(fn) {
      const cache = new Map();
      return (arg) => {
        if (cache.has(arg)) return cache.get(arg);
        const result = fn(arg);
        cache.set(arg, result);
        return result;
      };
    }

    let calls = 0;
    const expensive = (n) => { calls++; return n * 2; };
    const fast = memoize(expensive);

    fast(5);
    fast(5);
    fast(5);
    console.log(calls);
    ```
  </div>
  </slot>
  <slot name='explanation'>
  <div className="explanation">
